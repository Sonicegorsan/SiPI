60 QJsonObject response = responseDoc.object();
61
62 QString status = response["status"].toString();
63
64 if (status == "OK") {
65 if (response.contains("messages")) {
66 QJsonArray messagesArray = response["messages"].toArray();
67 messageView->clear();
68 for (const QJsonValue &value : messagesArray) {
69 QJsonObject messageObj = value.toObject();
70 QString sender = messageObj["sender"].toString();
71 QString content = messageObj["content"].toString();

73 messageView->append(QString("[%1] %2: %3").arg(timestamp, sender, content));
