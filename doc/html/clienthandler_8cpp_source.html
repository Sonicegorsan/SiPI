66void ClientHandler::processData(const QByteArray &data) {
67 QJsonDocument doc = QJsonDocument::fromJson(data);
68 if (!doc.isObject()) {
69 qWarning() << "Получены некорректные данные";
70 return;
71 }
72
73 QJsonObject request = doc.object();
74 QString command = request["command"].toString();
75
76 // Обработка команды LOGIN
77 if (command == "LOGIN") {
78 QString username = request["username"].toString();
79 QString passwordHash = request["password"].toString();
80
81 QSqlQuery query(db);
82 query.prepare("SELECT password_hash, role FROM users WHERE username = :username");
83 query.bindValue(":username", username);
84
85 if (!query.exec()) {
86 qWarning() << "Ошибка запроса к базе данных:" << query.lastError().text();
87 return;
88 }
89
90 if (query.next()) {
91 QString storedHash = query.value("password_hash").toString();
92 QString role = query.value("role").toString();
93 if (storedHash == passwordHash) {
94 this->username = username;
95
96 QJsonObject response;
97 response["status"] = "OK";
98 response["message"] = "Успешный вход";
99 response["role"] = role;
100
101 sendResponse(QJsonDocument(response).toJson());
102 qDebug() << "Пользователь" << username << "успешно вошёл как" << role;
103 } else {
104 QJsonObject response;
105 response["status"] = "ERROR";
106 response["message"] = "Неверный пароль";
107 sendResponse(QJsonDocument(response).toJson());
108 qWarning() << "Неверный пароль для пользователя" << username;
109 }
110 } else {
111 QJsonObject response;
112 response["status"] = "ERROR";
113 response["message"] = "Пользователь не найден";
114 sendResponse(QJsonDocument(response).toJson());
115 qWarning() << "Пользователь не найден:" << username;
116 }
117 }
118 // Обработка команды REGISTER
119 else if (command == "REGISTER") {
120 QString username = request["username"].toString();
121 QString passwordHash = request["password"].toString();
122
123 // Проверяем, существует ли уже такой пользователь
124 QSqlQuery query(db);
125 query.prepare("SELECT id FROM users WHERE username = :username");
126 query.bindValue(":username", username);
127
128 if (!query.exec()) {
129 qWarning() << "Ошибка запроса к базе данных:" << query.lastError().text();
130 return;
131 }
132
133 if (query.next()) {
134 QJsonObject response;
135 response["status"] = "ERROR";
136 response["message"] = "Пользователь уже существует";
137 sendResponse(QJsonDocument(response).toJson());
138 qWarning() << "Попытка регистрации существующего пользователя:" << username;
139 } else {
140 // Вставляем нового пользователя
141 QSqlQuery insertQuery(db);
142 insertQuery.prepare("INSERT INTO users (username, password_hash, role) VALUES (:username, :password_hash, 'employee')");
143 insertQuery.bindValue(":username", username);
144 insertQuery.bindValue(":password_hash", passwordHash);
145
146 if (insertQuery.exec()) {
147 QJsonObject response;
148 response["status"] = "OK";
149 response["message"] = "Регистрация успешна";
150 sendResponse(QJsonDocument(response).toJson());
151 qDebug() << "Пользователь зарегистрирован:" << username;
152 } else {
153 QJsonObject response;
154 response["status"] = "ERROR";
155 response["message"] = "Ошибка регистрации";
156 sendResponse(QJsonDocument(response).toJson());
157 qWarning() << "Ошибка регистрации пользователя:" << insertQuery.lastError().text();
158 }
159 }
160 }
161 // Обработка команды GET_ROOMS
162 else if (command == "GET_ROOMS") {
163 QSqlQuery query(db);
164 query.prepare("SELECT id, name FROM rooms");
165
166 if (query.exec()) {
167 QJsonArray roomsArray;
168 while (query.next()) {
169 QJsonObject roomObj;
170 roomObj["id"] = query.value("id").toInt();
171 roomObj["name"] = query.value("name").toString();
172 roomsArray.append(roomObj);
173 }
174
175 QJsonObject response;
176 response["status"] = "OK";
177 response["rooms"] = roomsArray;
178 sendResponse(QJsonDocument(response).toJson());
179 } else {
180 QJsonObject response;
181 response["status"] = "ERROR";
182 response["message"] = "Ошибка получения списка комнат";
183 sendResponse(QJsonDocument(response).toJson());
184 qWarning() << "Ошибка получения списка комнат:" << query.lastError().text();
185 }
186 }
187 // Обработка команды SEND_MESSAGE
188 else if (command == "SEND_MESSAGE") {
189 int roomId = request["room_id"].toInt();
190 QString content = request["content"].toString();
191
192 if (this->username.isEmpty()) {
193 QJsonObject response;
194 response["status"] = "ERROR";
195 response["message"] = "Пользователь не аутентифицирован";
196 sendResponse(QJsonDocument(response).toJson());
197 return;
198 }
199
200 // Получаем ID пользователя
201 QSqlQuery query(db);
202 query.prepare("SELECT id FROM users WHERE username = :username");
203 query.bindValue(":username", this->username);
204
205 if (query.exec() && query.next()) {
206 int senderId = query.value("id").toInt();
207
208 // Вставляем сообщение
209 QSqlQuery insertQuery(db);
210 insertQuery.prepare("INSERT INTO messages (room_id, sender_id, content) VALUES (:room_id, :sender_id, :content)");
211 insertQuery.bindValue(":room_id", roomId);
212 insertQuery.bindValue(":sender_id", senderId);
213 insertQuery.bindValue(":content", content);
214
215 if (insertQuery.exec()) {
216 QJsonObject response;
217 response["status"] = "OK";
218 response["message"] = "Сообщение отправлено";
219 sendResponse(QJsonDocument(response).toJson());
220 qDebug() << "Сообщение от" << this->username << "в комнату" << roomId;
221 } else {
222 QJsonObject response;
223 response["status"] = "ERROR";
224 response["message"] = "Ошибка отправки сообщения";
225 sendResponse(QJsonDocument(response).toJson());
226 qWarning() << "Ошибка отправки сообщения:" << insertQuery.lastError().text();
227 }
228 } else {
229 QJsonObject response;
230 response["status"] = "ERROR";
231 response["message"] = "Ошибка получения информации о пользователе";
232 sendResponse(QJsonDocument(response).toJson());
233 qWarning() << "Ошибка получения информации о пользователе:" << query.lastError().text();
234 }
235 }
236 // ... другие обработчики команд ...
237 // Остальной код метода оставляется без изменений, но с такими же подробными комментариями
238 // ...
239
240 // В конце метода обрабатываем неизвестные команды
241 else {
242 QJsonObject response;
243 response["status"] = "ERROR";
244 response["message"] = "Неизвестная команда";
245 sendResponse(QJsonDocument(response).toJson());
246 qWarning() << "Получена неизвестная команда";
247 }
248}
