91void MainWindow::onReadyRead() {
92 QByteArray responseData = socket->readAll();
93 QJsonDocument responseDoc = QJsonDocument::fromJson(responseData);
94 QJsonObject response = responseDoc.object();
95
96 QString status = response["status"].toString();
97
98 if (status == "OK") {
99 if (response.contains("rooms")) {
100 // Обработка списка комнат
101 QJsonArray roomsArray = response["rooms"].toArray();
102 chatTree->clear();
103 for (const QJsonValue &value : roomsArray) {
104 QJsonObject roomObj = value.toObject();
105 QTreeWidgetItem *roomItem = new QTreeWidgetItem(chatTree);
106 roomItem->setText(0, roomObj["name"].toString());
107 roomItem->setData(0, Qt::UserRole, roomObj["id"].toInt());
108 }
109 } else if (response.contains("tasks")) {
110 // Обработка списка задач
111 QJsonArray tasksArray = response["tasks"].toArray();
112 onTasksReceived(tasksArray);
113 }
114 } else {
115 QString message = response["message"].toString();
116 qWarning() << "Ошибка от сервера:" << message;
117 }
